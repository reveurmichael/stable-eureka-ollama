def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_reward = {}

    if not self.game_over:
        if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
            reward += REACH_GOAL_REWARD
            individual_reward['reach_goal'] = REACH_GOAL_REWARD
        else:
            if action[0] != 0 and action[1] != 0 and action[2] != 0 and action[3] != 0:
                reward -= STUCK_PENALTY
                individual_reward['stuck_penalty'] = -STUCK_PENALTY

    if pos[0] < 0 or self.game_over:
        reward -= FALL_DOWN_PENALTY
        individual_reward['fall_down_penalty'] = -FALL_DOWN_PENALTY

    return np.exp(reward / TEMPERATURE), individual_reward
