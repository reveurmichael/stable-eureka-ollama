def compute_reward(self, pos, action, state):
    distance_to_goal = (self.hull.position[0] - VIEWPORT_W / SCALE) / TERRAIN_STEP
    if self.game_over or pos[0] < 0:
        return -10.0, {'failure': -1.0}

    reward_speed = np.tanh(action[0]) * 2.0 + np.tanh(action[1]) * 4.0
    reward_distance = distance_to_goal ** 2
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        return 10.0, {'success': 5.0}
    else:
        return reward_speed + reward_distance, {'speed': reward_speed, 'distance': reward_distance}
