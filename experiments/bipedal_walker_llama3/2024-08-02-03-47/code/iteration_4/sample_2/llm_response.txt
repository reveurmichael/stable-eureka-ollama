def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_reward = {"position": 0.0, "speed": 0.0, "fell": 0.0, "stuck": 0.0}

    # Reward for position progress
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 10.0 / self.steps_to_end
        individual_reward["position"] = 10.0 / self.steps_to_end

    # Penalty for falling down
    if self.game_over or pos[0] < 0:
        reward -= 10.0
        individual_reward["fell"] = -10.0

    # Reward for forward movement
    if not control_speed and action[0] > 0:
        reward += 0.1 * abs(action[0])
        individual_reward["speed"] += 0.1 * abs(action[0])

    return reward, individual_reward
