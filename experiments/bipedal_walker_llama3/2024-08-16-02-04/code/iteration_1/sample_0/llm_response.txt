def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_rewards = {}
    
    # Reward for reaching the end
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 100.0
    
    # Penalty for not moving forward
    if action[0] == 0 and action[1] == 0 and action[2] == 0 and action[3] == 0:
        reward -= 10.0
    
    # Reward for staying upright
    if state[-5] > 0.5 or state[-11] > 0.5:
        reward += 5.0
    
    individual_rewards['reached_end'] = reward
    individual_rewards['moved_forward'] = -reward
    individual_rewards['stayed_upright'] = reward
    
    return np.exp(reward / 10.0), individual_rewards
