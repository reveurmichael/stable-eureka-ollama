def compute_reward(self, pos, action, state):
    distance = pos[0] / (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP
    speed = np.sum(np.square([state[i*2+1] for i in range(5)]))
    
    reward = np.exp((distance - 0.5) / 0.3)
    individual_reward = {'distance': reward, 'speed': -np.exp(-speed / (SPEED_HIP * SPEED_KNEE)), 'falling_down': -1000 if pos[1] > 10 else 0}
    
    return reward, individual_reward
