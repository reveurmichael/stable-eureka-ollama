def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_rewards = {}
    
    # Reward for reaching the end of the terrain
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 10.0
        individual_rewards['end_reached'] = 10.0
    
    # Penalty for falling down or getting stuck
    if self.game_over or pos[0] < 0:
        reward -= 5.0
        individual_rewards['fallen_or_stuck'] = -5.0
    
    # Reward for moving forward
    reward += (state[3] + state[2]) / FPS * 0.1
    individual_rewards['forward_movement'] = (state[3] + state[2]) / FPS * 0.1
    
    return reward, individual_rewards
