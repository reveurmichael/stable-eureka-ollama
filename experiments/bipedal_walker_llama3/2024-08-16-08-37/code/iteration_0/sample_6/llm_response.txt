def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_reward = {"pos": 0.0, "action": 0.0}

    # Penalties for falling or getting stuck
    if self.game_over:
        reward -= 1.0
        individual_reward["pos"] -= 1.0

    # Reward for reaching the end of the terrain
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 10.0
        individual_reward["pos"] += 10.0

    # Encourage movement forward and avoid obstacles
    reward -= abs(action[0]) + abs(action[1]) + abs(action[2]) + abs(action[3])
    individual_reward["action"] -= (abs(action[0]) + abs(action[1]) + abs(action[2]) + abs(action[3]))

    return reward, individual_reward
