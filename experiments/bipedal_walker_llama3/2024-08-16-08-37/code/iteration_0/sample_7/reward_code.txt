def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_reward = {}

    # Reward for reaching the end of the terrain
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 10.0
        individual_reward['reached_end'] = 10.0

    # Penalty for falling down or getting stuck
    if self.game_over or pos[0] < 0:
        reward -= 5.0
        individual_reward['fell_down'] = -5.0

    # Reward for moving forward
    reward += (pos[0] / TERRAIN_LENGTH) * 2.0
    individual_reward['moved_forward'] = (pos[0] / TERRAIN_LENGTH) * 2.0

    return reward, individual_reward
