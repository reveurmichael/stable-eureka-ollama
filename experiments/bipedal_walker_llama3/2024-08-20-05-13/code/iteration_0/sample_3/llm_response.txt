        reward = -1.0

    else:
        # Encourage forward movement
        movement_reward = np.exp((pos[0] + VIEWPORT_W / SCALE / 5) / TERRAIN_LENGTH)
        # Punish falling or getting stuck
        fall_penalty = max(-np.exp(hull_angle), 0)
        reward = movement_reward - fall_penalty
