def compute_reward(self, pos, action, state):
    if not self.game_over and pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        # Reward for reaching the end of the terrain
        reward = 10.0
    elif not self.game_over:
        # Penalty for not falling, but also not moving forward
        reward = -2.0
    else:
        # Penalties for falling or getting stuck
        reward = -5.0

    individual_reward = {'end_reached': 10.0 if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP else 0.0,
                          'moving_forward': -2.0 if not self.game_over and state[3] < 0.1 else 0.0,
                          'fell_or_stuck': -5.0 if self.game_over else 0.0}

    return reward, individual_reward
