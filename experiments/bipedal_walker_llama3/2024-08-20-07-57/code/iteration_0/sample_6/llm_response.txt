def compute_reward(self, pos, action, state):
    reward = 0.0
    individual_reward = {}

    # Encourage reaching the end of the terrain
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 10.0
        individual_reward['end_reached'] = 10.0

    # Penalty for not moving forward
    if action[0] != 0 and action[2] != 0:  # Check for non-zero hip actions
        reward -= 1.0
        individual_reward['movement_penalty'] = -1.0

    # Reward for reaching the end of the terrain quickly
    if pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward -= (state[-2] + state[-4])  # Normalize velocity values
        individual_reward['speed_penalty'] = -(state[-2] + state[-4])

    return reward, individual_reward
