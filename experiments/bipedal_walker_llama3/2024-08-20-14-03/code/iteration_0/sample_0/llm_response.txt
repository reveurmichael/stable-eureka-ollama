def compute_reward(self, pos, action, state):
    distance_to_goal = (VIEWPORT_W / SCALE) - pos[0]
    
    if pos[0] > TERRAIN_LENGTH:
        return 100.0, {'distance_to_goal': 0.0}
    elif not self.game_over and abs(action[2]) > 0.5:  # if moving forward
        reward = np.exp((VIEWPORT_W / SCALE) - pos[0] - 1e-6) + np.exp(1 - distance_to_goal)
        return float(reward), {'distance_to_goal': distance_to_goal}
    else:
        reward = np.exp(1 - distance_to_goal)
        return float(reward), {'distance_to_goal': distance_to_goal}
