    x_velocity = state[2]

    distance_to_goal = abs(pos[0] - TERRAIN_LENGTH)
    fall_penalty = 1.0 if pos[1] < 0 else 0.0
    reward = -np.exp(-distance_to_goal) - fall_penalty + np.exp(x_velocity)
