def compute_reward(self, pos, action, state):
    reward = -1.0
    if not self.game_over and pos[0] > (TERRAIN_LENGTH - TERRAIN_GRASS) * TERRAIN_STEP:
        reward += 10.0
    for i in range(4):
        if not self.legs[i].ground_contact:
            reward -= 1.0
    return reward, {'reached_goal': reward > 0}
