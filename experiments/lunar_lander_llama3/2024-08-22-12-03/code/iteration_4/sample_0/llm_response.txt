def compute_reward(self, state, m_power, s_power, terminated):
    if self.game_over:
        return -1.0, {'success': 1.0}
    elif abs(state[0]) >= 1.0 or not self.lander.awake:
        return -10.0, {'crash': 1.0}
    elif state[5] > 30.0: # If the lander is rotating too fast
        return -2.0, {'over_rotate': 1.0}
    else:
        reward = -1.0 + (m_power / MAX_ENGINE_POWER) + (s_power / MAX_SIDE_ENGINE_POWER)
        return reward, {'reward': reward}
